function tablify.tablifySingleDict (o, opts)

  • description and source-code
    tablifySingleDict = function (o, opts) {
  var arr, k, v;
  arr = [];
  for (k in o) {
    v = o[k];
    arr.push([k, v]);
  }
  arr.sort(function(r1, r2) {
    return r1[0].localeCompare(r2[0]);
  });
  return exports.tablifyArrays(arr, opts);
}
